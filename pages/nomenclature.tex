\renewcommand{\nompreamble}{The following list describes several symbols and abbreviations used in the report. All units follow the SI unit system.}

%---------------------------------------------------------------------------------
%%-------------------------------- KINEMATICS ------------------------------------
%---------------------------------------------------------------------------------

\nomenclature[R, 01]{$n$}{Number of links}
\nomenclature[R, 02]{$n_c$}{Number of obstacles in contact}
\nomenclature[R, 03]{$N$}{Number of generalized coordinates}
\nomenclature[R, 04]{$m$}{Link mass}
\nomenclature[R, 05]{$l$}{Link length}

\nomenclature[R, 06]{$\mathbf{q}$}{Generalized coordinates}
\nomenclature[R, 07]{$\phi_i$}{Joint angle of link $i$ relative to preceding link}
\nomenclature[R, 08]{$\theta_i$}{Angle of link $i$ relative to the base frame}

\nomenclature[R, 10]{$(x_0, y_0)$}{Position of tail in base frame}
\nomenclature[R, 11]{$(x_i, y_i)$}{Position of endpoint of link $i$ in base frame $b$}
\nomenclature[R, 12]{$(x_{c,i}, y_{c,i})$}{Position of contact point on link $i$ in base frame $b$}
\nomenclature[R, 13]{$k$}{Distance from joint to contact point on proceeding link}
\nomenclature[R, 14]{$\mathbf{T_{bi}}$}{Transformation matrix from base frame to frame of link $i$}
\nomenclature[R, 15]{$\mathbf{T_{bci}}$}{Transformation matrix from base frame to frame of contact point on link $i$}
\nomenclature[R, 16]{$\mathbf{v_i}$}{Velocity of endpoint of link $i$ in base frame $b$}
\nomenclature[R, 17]{$\mathbf{v_{c,i}}$}{Velocity of contact point on link $i$ in base frame $b$}

\nomenclature[R, 18]{$\mathbf{J}_t$}{Task Jacobian}
\nomenclature[R, 19]{$\mathbf{J}_c$}{Constraint Jacobian}

\nomenclature[R, 20]{$\mathbf{r}_t$}{Task space coordinates}
\nomenclature[R, 21]{$\mathbf{r}_c$}{Constraint space coordinates}

%---------------------------------------------------------------------------------
%%-------------------------------- DYNAMICS --------------------------------------
%---------------------------------------------------------------------------------

\nomenclature[D, 01]{$\boldsymbol{\tau}$}{Generalized torques}

\nomenclature[D, 05]{$\mathbf{M}$}{Mass matrix}
\nomenclature[D, 06]{$\mathbf{C}$}{Coriolis matrix}
\nomenclature[D, 07]{$\mathbf{g}$}{Gravity matrix}

\nomenclature[D, 08]{$L$}{Lagrangian}
\nomenclature[D, 09]{$K, P$}{Kinetic and potential energy}
\nomenclature[D, 10]{$I$}{Moment of inertia of rod}

\nomenclature[D, 11]{$\mathbf{E}_F$}{Task frame axes of force constraints}
\nomenclature[D, 12]{$\mathbf{E}_P$}{Task frame axes of position constraints}



%---------------------------------------------------------------------------------

\nomenclature[C, 03]{$\boldsymbol{\tau}_c$}{Joint motor control torques}
\nomenclature[C, 04]{$\boldsymbol{\tau}_P$}{Joint torques for desired position control}
\nomenclature[C, 05]{$\boldsymbol{\tau}_F$}{Joint torques for desired force control}
